using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApp1
{
    class Program
    {
        static void Main(string[] args)
        {
            int counters = 0;

            for(var i = 1; i<= 21 ; i++)
            {
                string ingr = Console.ReadLine();

                counters = i - 1;
                if(ingr != "Bake!")
                {
                    Console.WriteLine($"Adding ingredient {ingr}.");
                }
                else
                {
                    Console.WriteLine($"Preparing cake with {counters} ingredients.");break;
                }
            }
        }
    }
}
